ALTER PROCEDURE [dbo].[SP_CHECK_USERNAME]
    @USERNAME VARCHAR(50) 
AS
BEGIN
    SET NOCOUNT ON;

    IF EXISTS (SELECT 1 FROM sys.database_principals WHERE name = @USERNAME)
    BEGIN
        RETURN 1 
    END
	IF EXISTS (SELECT 1 FROM LINK1.NGANHANG.sys.database_principals WHERE name = @USERNAME)
    BEGIN
        RETURN 1
    END
	RETURN 0
END
